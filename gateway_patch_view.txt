    18	app = FastAPI()

    20	BROKERAGE_LEDGER = "/home/rick/pet_alpha/brokerage_ledger.json"
    21	
    22	def get_latest_brokerage_record():
    23	    if not os.path.exists(BROKERAGE_LEDGER): return {}
    24	    with open(BROKERAGE_LEDGER, "r") as f:
    25	        lines = [line.strip() for line in f if line.strip()]
    26	    return json.loads(lines[-1]) if lines else {}

    28	@app.get("/dashboard", response_class=HTMLResponse)

    30	    latest_record = get_latest_brokerage_record()

    34	    return templates.TemplateResponse(

    48	@app.get("/clients", response_class=HTMLResponse)
    49	async def clients_page(request: Request):
    50	    records = load_brokerage_records()
    51	
    52	    latest_by_client = {}
    53	    for r in records:
    54	        name = r.get("brokerage_name", "Unknown Brokerage")
    55	        latest_by_client[name] = r
    56	
    57	    records = list(latest_by_client.values())
    58	    records.sort(key=lambda x: x.get("brokerage_name", ""))
    59	
    60	    enriched_records = []
    61	    for r in records:
    62	        enriched_records.append({
    63	            "record": r,
    64	            "analysis": analyze_client_record(r)
    65	        })
    66	
    67	    return templates.TemplateResponse(
    68	        request=request,
    69	        name="clients.html",
    70	        context={"clients": enriched_records}
    71	    )

   114	                )
